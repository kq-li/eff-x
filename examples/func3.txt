compose: (int -> int) -> (int -> int) -> int -> int = \f: int -> int. \g: int -> int. \x: int. {
  x: int = g x;
  x: int = f x;
  return x;
};

add5: int -> int = \x: int. x + 5;
times2: int -> int = \x: int. x * 2;
a: int = compose add5 times2 3;
_: unit !output = print a;
b: int = compose times2 add5 3;
_: unit !output = print b;
